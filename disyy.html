    第五题：打印100以内数字，3和5整除以foobar代替，3整除foo代替，5整除bar代替

            <?php
            for($i = 1; $i <= 100; $i++) {
                if($i % 3 === 0 && $i % 5 === 0){
                    echo 'foobar ';
                }else if($i % 3 === 0){
                    echo 'foo ';
                }else if($i % 5 === 0){
                    echo 'bar ';
                }else{
                    echo $i . ' ';
                }
            }
        ?>
